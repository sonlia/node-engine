   617‚Üí   617‚Üí   *

   619‚Üí   619‚Üí   *   - Type hierarchy (int ‚Üí number ‚Üí * subtype relationships)
   620‚Üí   620‚Üí   *   - Wildcard matching (0, null, undefined, "", "*")
   621‚Üí   621‚Üí   *   - EVENT ‚Üî ACTION cross-matching
   622‚Üí   622‚Üí   *   - Comma-separated multi-type slots
   623‚Üí   623‚Üí   *   - Fallback to string comparison for unregistered types
   624‚Üí   624‚Üí   *
   625‚Üí   625‚Üí   * If SlotTypes is not yet loaded (circular dep guard), falls back to
   626‚Üí   626‚Üí   * the legacy string-comparison logic.
   627‚Üí   627‚Üí   */
   628‚Üí   628‚Üí  static isValidConnection(typeA, typeB) {
   629‚Üí   629‚Üí    // Try the new SlotTypes-based system first
   630‚Üí   630‚Üí    if (LiteGraph.SlotTypes) {
   631‚Üí   631‚Üí      return LiteGraph.SlotTypes.isValidConnection(typeA, typeB);
   632‚Üí   632‚Üí    }
   633‚Üí   633‚Üí    // Legacy fallback (mirrors original logic)
   634‚Üí   634‚Üí    if (typeA === "" || typeA === "*") typeA = 0;
   635‚Üí   635‚Üí    if (typeB === "" || typeB === "*") typeB = 0;
   636‚Üí   636‚Üí    if (
   637‚Üí   637‚Üí      typeA === 0 || typeA === null || typeA === undefined ||
   638‚Üí   638‚Üí      typeB === 0 || typeB === null || typeB === undefined
   639‚Üí   639‚Üí    ) {
   640‚Üí   640‚Üí      return true;
   641‚Üí   641‚Üí    }
   642‚Üí   642‚Üí    if (typeA === typeB) return true;
   643‚Üí   643‚Üí    const strA = String(typeA).toLowerCase();
   644‚Üí   644‚Üí    const strB = String(typeB).toLowerCase();
   645‚Üí   645‚Üí    if (strA === strB) return true;
   646‚Üí   646‚Üí    if (strA.indexOf(",") !== -1 || strB.indexOf(",") !== -1) {
   647‚Üí   647‚Üí      const listA = strA.split(",");
   648‚Üí   648‚Üí      const listB = strB.split(",");
   649‚Üí   649‚Üí      for (const a of listA) for (const b of listB) {
   650‚Üí   650‚Üí        if (a.trim() === b.trim()) return true;
   651‚Üí   651‚Üí      }
   652‚Üí   652‚Üí    }
   653‚Üí   653‚Üí    return false;
   654‚Üí   654‚Üí  }

   669‚Üí   669‚Üí  /**
   670‚Üí   670‚Üí   * Wrapper to load files (from URL using fetch, or from File/Blob using FileReader).
   671‚Üí   671‚Üí   * Restored original (url, type, on_complete, on_error) signature with
   672‚Üí   672‚Üí   * full response-type support (text/arraybuffer/json/blob), proxy prefixing,
   673‚Üí   673‚Üí   * and FileReader branches for File/Blob inputs. Returns Promise (URL) or
   674‚Üí   674‚Üí   * FileReader (File/Blob).
   675‚Üí   675‚Üí   */
   676‚Üí   676‚Üí  static fetchFile(url, type, on_complete, on_error) {
   677‚Üí   677‚Üí    type = type || "text";
   678‚Üí   678‚Üí    if (!url) return null;
   679‚Üí   679‚Üí
   680‚Üí   680‚Üí    if (url.constructor === String) {
   681‚Üí   681‚Üí      // Apply proxy if configured (e.g. for CORS workarounds).
   682‚Üí   682‚Üí      if (url.substr(0, 4) === "http" && LiteGraph.proxy) {
   683‚Üí   683‚Üí        url = LiteGraph.proxy + url.substr(url.indexOf(":") + 3);
   684‚Üí   684‚Üí      }
   685‚Üí   685‚Üí      return fetch(url)
   686‚Üí   686‚Üí        .then(function (response) {
   687‚Üí   687‚Üí          if (!response.ok) throw new Error("File not found");
   688‚Üí   688‚Üí          if (type === "arraybuffer") return response.arrayBuffer();
   689‚Üí   689‚Üí          else if (type === "text" || type === "string") return response.text();
   690‚Üí   690‚Üí          else if (type === "json") return response.json();
   691‚Üí   691‚Üí          else if (type === "blob") return response.blob();
   692‚Üí   692‚Üí        })
   693‚Üí   693‚Üí        .then(function (data) {
   694‚Üí   694‚Üí          if (on_complete) on_complete(data);
   695‚Üí   695‚Üí        })
   696‚Üí   696‚Üí        .catch(function (error) {
   697‚Üí   697‚Üí          console.error("error fetching file:", url);
   698‚Üí   698‚Üí          if (on_error) on_error(error);
   699‚Üí   699‚Üí        });
   700‚Üí   700‚Üí    } else if (url.constructor === File || url.constructor === Blob) {
   701‚Üí   701‚Üí      const reader = new FileReader();
   702‚Üí   702‚Üí      reader.onload = function (e) {
   703‚Üí   703‚Üí        let v = e.target.result;
   704‚Üí   704‚Üí        if (type === "json") v = JSON.parse(v);
   705‚Üí   705‚Üí        if (on_complete) on_complete(v);
   706‚Üí   706‚Üí      };
   707‚Üí   707‚Üí      if (type === "arraybuffer") return reader.readAsArrayBuffer(url);
   708‚Üí   708‚Üí      else if (type === "text" || type === "json") return reader.readAsText(url);
   709‚Üí   709‚Üí      else if (type === "blob") return reader.readAsBinaryString(url);
   710‚Üí   710‚Üí      return reader;
   711‚Üí   711‚Üí    }
   712‚Üí   712‚Üí    return null;
   713‚Üí   713‚Üí  }

   740‚Üí   740‚Üí   */
   741‚Üí   741‚Üí  static extendClass(target, origin) {
   742‚Üí   742‚Üí    // Copy own static properties from origin to target (skip ones target
   743‚Üí   743‚Üí    // already owns, mirroring the original `hasOwnProperty` guard).
   744‚Üí   744‚Üí    for (const i in origin) {
   745‚Üí   745‚Üí      if (target[i] === undefined) {
   746‚Üí   746‚Üí        target[i] = origin[i];
   747‚Üí   747‚Üí      }
   748‚Üí   748‚Üí    }
   749‚Üí   749‚Üí    // Copy prototype properties (including getters/setters) so subclassing
   750‚Üí   750‚Üí    // via `LiteGraph.extendClass(Sub, LGraphNode)` actually inherits methods.
   751‚Üí   751‚Üí    if (origin.prototype) {
   752‚Üí   752‚Üí      for (const i in origin.prototype) {
   753‚Üí   753‚Üí        if (!Object.prototype.hasOwnProperty.call(origin.prototype, i))
   754‚Üí   754‚Üí          continue;
   755‚Üí   755‚Üí        if (Object.prototype.hasOwnProperty.call(target.prototype, i))
   756‚Üí   756‚Üí          continue;
   757‚Üí   757‚Üí        const getter = origin.prototype.__lookupGetter__(i);
   758‚Üí   758‚Üí        const setter = origin.prototype.__lookupSetter__(i);
   759‚Üí   759‚Üí        if (getter) {
   760‚Üí   760‚Üí          target.prototype.__defineGetter__(i, getter);
   761‚Üí   761‚Üí        }
   762‚Üí   762‚Üí        if (setter) {
   763‚Üí   763‚Üí          target.prototype.__defineSetter__(i, setter);
   764‚Üí   764‚Üí        }
   765‚Üí   765‚Üí        if (!getter && !setter) {
   766‚Üí   766‚Üí          target.prototype[i] = origin.prototype[i];
   767‚Üí   767‚Üí        }
   768‚Üí   768‚Üí      }
   769‚Üí   769‚Üí    }
   770‚Üí   770‚Üí  }
