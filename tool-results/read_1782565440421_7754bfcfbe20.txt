    31→/**
    32→ * Get the mouse button from an event.
    33→ * PointerEvent may not set e.which, so we fall back to e.button.
    34→ * Returns: 1=left, 2=middle, 3=right
    35→ */
    36→function getMouseButton(e) {
    37→    if (e.which !== undefined && e.which !== 0) return e.which;
    38→    // e.button: 0=left, 1=middle, 2=right
    39→    if (e.button === 0) return 1;
    40→    if (e.button === 1) return 2;
    41→    if (e.button === 2) return 3;
    42→    return 0;
    43→}

   313→    /** assigns a canvas */
   314→    setCanvas(canvas, skip_events) {
   315→        if (canvas) {
   316→            if (canvas.constructor === String) {
   317→                canvas = document.getElementById(canvas);
   318→                if (!canvas) {
   319→                    throw "Error creating LiteGraph canvas: Canvas not found";
   320→                }
   321→            }
   322→        }
   323→
   324→        if (canvas === this.canvas) return;
   325→
   326→        if (!canvas && this.canvas) {
   327→            if (!skip_events) {
   328→                this.unbindEvents();
   329→            }
   330→        }
   331→
   332→        this.canvas = canvas;
   333→        this.ds.element = canvas;
   334→
   335→        if (!canvas) return;
   336→
   337→        canvas.className += " lgraphcanvas";
   338→        canvas.data = this;
   339→        canvas.tabindex = "1";
   340→
   341→        // bg canvas: used for non changing stuff
   342→        this.bgcanvas = null;
   343→        if (!this.bgcanvas) {
   344→            this.bgcanvas = document.createElement("canvas");
   345→            this.bgcanvas.width = this.canvas.width;
   346→            this.bgcanvas.height = this.canvas.height;
   347→        }
   348→
   349→        if (canvas.getContext == null) {
   350→            if (canvas.localName !== "canvas") {
   351→                throw (
   352→                    "Element supplied for LGraphCanvas must be a <canvas> element, you passed a " +
   353→                    canvas.localName
   354→                );
   355→            }
   356→            throw "This browser doesn't support Canvas";
   357→        }
   358→
   359→        const ctx = (this.ctx = canvas.getContext("2d"));
   360→        if (ctx == null) {
   361→            if (!canvas.webgl_enabled) {
   362→                console.warn("This canvas seems to be WebGL, enabling WebGL renderer");
   363→            }
   364→            this.enableWebGL();
   365→        }
   366→
   367→        if (!skip_events) {
   368→            this.bindEvents();
   369→        }
   370→    }

   391→        if (this._events_binded) {

   410→        canvas.addEventListener("contextmenu", this._doNothing);

   412→            "DOMMouseScroll",
   413→            this._mousewheel_callback,
   414→            false

   427→        canvas.addEventListener("drop", this._ondrop_callback, false);

   519→    // ==================== MOUSE HANDLING ====================
   520→
   521→    processMouseDown(e) {
   522→        if (this.set_canvas_dirty_on_mouse_event) this.dirty_canvas = true;
   523→
   524→        if (!this.graph) return;
   525→
   526→        this.adjustMouseEvent(e);
   527→
   528→        const ref_window = this.getCanvasWindow();
   529→        LGraphCanvas.active_canvas = this;
   530→
   531→        const x = e.clientX;
   532→        const y = e.clientY;
   533→
   534→        this.ds.viewport = this.viewport;
   535→        const is_inside =
   536→            !this.viewport ||
   537→            (this.viewport &&
   538→                x >= this.viewport[0] &&
   539→                x < this.viewport[0] + this.viewport[2] &&
   540→                y >= this.viewport[1] &&
   541→                y < this.viewport[1] + this.viewport[3]);
   542→
   543→        // move mouse move event to the window in case it drags outside of the canvas
   544→        if (!this.options.skip_events) {
   545→            pointerListenerRemove(this.canvas, "move", this._mousemove_callback);
   546→            pointerListenerAdd(
   547→                ref_window.document,
   548→                "move",
   549→                this._mousemove_callback,
   550→                true
   551→            );
   552→            pointerListenerAdd(
   553→                ref_window.document,
   554→                "up",
   555→                this._mouseup_callback,
   556→                true
   557→            );
   558→        }
   559→
   560→        if (!is_inside) return;
   561→
   562→        let node = this.graph.getNodeOnPos(
   563→            e.canvasX,
   564→            e.canvasY,
   565→            this.visible_nodes,
   566→            5
   567→        );
   568→        let skip_dragging = false;
   569→        let skip_action = false;
   570→        const now = getTime();
   571→        const is_primary = e.isPrimary === undefined || !e.isPrimary;
   572→        const is_double_click =
   573→            now - this.last_mouseclick < 300 && is_primary;
   574→
   575→        this.mouse[0] = e.clientX;
   576→        this.mouse[1] = e.clientY;
   577→        this.graph_mouse[0] = e.canvasX;
   578→        this.graph_mouse[1] = e.canvasY;
   579→        this.last_click_position = [this.mouse[0], this.mouse[1]];
   580→
   581→        if (this.pointer_is_down && is_primary) {
   582→            this.pointer_is_double = true;
   583→        } else {
   584→            this.pointer_is_double = false;
   585→        }
   586→        this.pointer_is_down = true;
   587→
   588→        this.canvas.focus();
   589→
   590→        LiteGraph.closeAllContextMenus(ref_window);
   591→
   592→        if (this.onMouse) {
   593→            if (this.onMouse(e) === true) return;
   594→        }
   595→
   596→        // LEFT BUTTON / single finger
   597→        if (getMouseButton(e) === 1 && !this.pointer_is_double) {
   598→            if (e.ctrlKey) {
   599→                this.dragging_rectangle = new Float32Array(4);
